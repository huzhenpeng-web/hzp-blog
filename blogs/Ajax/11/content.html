    # 一、JSON语法

        <!-- JSON语法注意事项
        1.属性名必须使用双引号包裹
        2.字符串类型的值必须使用双引号包裹
        3.JSON中不允许使用单引号表示字符串
        4.JSON中不能写注释
        5.JSON的最外层必须是对象或数组格式
        6.不能使用undefined或函数作为JSON的值
        作用:在计算机与网络之间存储和传输数据。
        JSON的本质:用字符串表示Javascript对象数据或数组数据
  -->
    <script>
        //json字符串和json对象的互换
        var jsonStr = '{"name":"胡振鹏","age":"20"}'
        console.log(jsonStr);
        var result = JSON.parse(jsonStr)
        console.log(result);
        //json对象转换为json字符串
        var json = JSON.stringify(result)
        console.log(json);
    </script>

    # 二、Json.parse函数的应用

    <!-- 把数据对象转化为字符串的过程 叫做序列化 调用JSON.stringify()操作 叫做JSON序列化 -->
<!-- 把字符串转化为数据对象的过程 叫做反序列化 调用JSON.parse()操作 叫做JSON反序列化 -->
    <script>
        var xhr = new XMLHttpRequest();
        xhr.open('GET', 'http:www.liulongbin.top:3006/api/getbooks')
        xhr.send()
        xhr.onreadystatechange = function() {
            if (xhr.readyState === 4 && xhr.status) {
                console.log(xhr.responseText);
                var jsonObj = JSON.parse(xhr.responseText)
                console.log(jsonObj); //转化为Json对象
            }
        }
    </script>
