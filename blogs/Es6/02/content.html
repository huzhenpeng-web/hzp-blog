    02-解构赋值

    # 一、数组解构

        <script>
        //数组解构允许我们按照一一对应的关系从数组中提取值然后将值赋值给变量
        let arr = [1, 2, 3]
        let [a, b, c, d, e] = arr;
        console.log(a);//1
        console.log(b);//2
        console.log(c);//3
        console.log(d);//undefined
        console.log(e);//undefined
    </script>

    # 二、对象解构

        <script>
        //按照一定模式 从数组中或对象中提取值,将提取出来的值赋值给另外的变量
        //对象解构允许我们使用变量的名字匹配对象的属性 匹配成功 将对象属性的值赋值给变量
        let person = {
            name: '张三',
            age: 20
        }
        let {
            name, //要和对象里的属性名保持一致
            age,
            classes //undefined
        } = person
        console.log(name);
        console.log(age);
        console.log(classes);
        console.log(person);
        console.log('-------------');
        //第二种
        let {
            name: studentName,
            age: myAge
        } = person
        console.log(studentName);
        console.log(myAge);
    </script>
